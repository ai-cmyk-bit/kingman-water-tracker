function doGet(e) {
  var sheet = SpreadsheetApp
    .getActiveSpreadsheet().getActiveSheet();
  var p = e.parameter;
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Startup Date/Time","Well #","Engine Hours Start",
      "Flow Meter Start","Blocks Irrigated",
      "Shutdown Date/Time","Shutdown Well #",
      "Flow Meter End","Engine Hours End",
      "Water Used (AF)","Engine Hours Used","Logged At"
    ]);
  }
  sheet.appendRow([
    p.startup_datetime, p.well, p.engine_hours_start,
    p.flow_meter_start, p.blocks,
    p.shutdown_datetime, p.shutdown_well,
    p.flow_meter_end, p.engine_hours_end,
    p.water_used, p.engine_hours_used,
    new Date().toISOString()
  ]);
  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
